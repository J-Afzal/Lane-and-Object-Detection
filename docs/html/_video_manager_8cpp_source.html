46 VideoManager::VideoManager(const std::vector<std::string>& p_commandLineArguments) : // NOLINT(readability-function-cognitive-complexity)
47 m_saveOutput(false)
48 {
49 std::string parsedInputVideoFilePath;
50 std::string parsedYoloFolderPath;
51 Globals::ObjectDetectorTypes parsedObjectDetectorTypes = Globals::ObjectDetectorTypes::NONE;
52 Globals::ObjectDetectorBackEnds parsedObjectDetectorBackEnds = Globals::ObjectDetectorBackEnds::CPU;
53 Globals::ObjectDetectorBlobSizes parsedObjectDetectorBlobSizes = Globals::ObjectDetectorBlobSizes::ONE;
54
55 uint32_t index = 0;
56
57 for (const std::string& argument : p_commandLineArguments)
58 {
59 if (argument == "-h" || argument == "--help")
60 {
61 std::cout << Globals::G_CLI_HELP_MESSAGE;
62 std::exit(1);
63 }
64
65 try
66 {
67 if (argument == "-i" || argument == "--input")
68 {
69 parsedInputVideoFilePath = p_commandLineArguments.at(index + 1);
70 }
71
72 if (argument == "-y" || argument == "--yolo-folder-path")
73 {
74 parsedYoloFolderPath = p_commandLineArguments.at(index + 1);
75 }
76
77 if (argument == "-o" || argument == "--object-detector-type")
78 {
79 if (p_commandLineArguments.at(index + 1) == "none")
80 {
81 parsedObjectDetectorTypes = Globals::ObjectDetectorTypes::NONE;
82 }
83
84 else if (p_commandLineArguments.at(index + 1) == "standard")
85 {
86 parsedObjectDetectorTypes = Globals::ObjectDetectorTypes::STANDARD;
87 }
88
89 else if (p_commandLineArguments.at(index + 1) == "tiny")
90 {
91 parsedObjectDetectorTypes = Globals::ObjectDetectorTypes::TINY;
92 }
93
94 else
95 {
96 std::cout << Globals::G_CLI_HELP_MESSAGE;
97 std::exit(1);
98 }
99 }
100
101 if (argument == "-b" || argument == "--object-detector-backend")
102 {
103 if (p_commandLineArguments.at(index + 1) == "cpu")
104 {
105 parsedObjectDetectorBackEnds = Globals::ObjectDetectorBackEnds::CPU;
106 }
107
108 else if (p_commandLineArguments.at(index + 1) == "cuda")
109 {
110 parsedObjectDetectorBackEnds = Globals::ObjectDetectorBackEnds::CUDA;
111 }
112
113 else
114 {
115 std::cout << Globals::G_CLI_HELP_MESSAGE;
116 std::exit(1);
117 }
118 }
119
120 if (argument == "-s" || argument == "--object-detector-blob-size")
121 {
122 parsedObjectDetectorBlobSizes = static_cast<Globals::ObjectDetectorBlobSizes>(std::stoi(p_commandLineArguments.at(index + 1)));
123 }
124 }
125
126 catch (...)
127 {
128 std::cout << Globals::G_CLI_HELP_MESSAGE;
129 std::exit(1);
130 }
131
132 index++;
133 }
134
135 // Check that the required arguments have been provided
136 if (parsedInputVideoFilePath.empty() || parsedYoloFolderPath.empty())
137 {
138 std::cout << Globals::G_CLI_HELP_MESSAGE;
139 std::exit(1);
140 }
141
142 SetProperties(parsedInputVideoFilePath, parsedYoloFolderPath, parsedObjectDetectorTypes, parsedObjectDetectorBackEnds, parsedObjectDetectorBlobSizes);
143 }

196 void VideoManager::RunLaneAndObjectDetector()
197 {
198 while (true)
199 {
200 m_performance.StartTimer();
201
202 if (!m_inputVideo.read(m_currentFrame))
203 {
204 break;
205 }
206
207 m_objectDetector.RunObjectDetector(m_currentFrame);
208
209 m_laneDetector.RunLaneDetector(m_currentFrame, m_objectDetector.GetInformation());
210
211 FrameBuilder::UpdateFrame(m_currentFrame, m_objectDetector.GetInformation(), m_laneDetector.GetInformation(), m_performance.GetInformation(), m_videoManagerInformation);
212
213 if (m_saveOutput)
214 {
215 m_outputVideo.write(m_currentFrame);
216 }
217
218 cv::imshow("frame", m_currentFrame);
219
220 switch (cv::waitKey(1))
221 {
222 case Globals::G_TOGGLE_RECORDING_KEY:
223 ToggleSaveOutput();
224 break;
225
226 case Globals::G_QUIT_KEY:
227 cv::destroyAllWindows();
228 return;
229
230 default:
231 break;
232 }
233
234 m_performance.EndTimer();
235 }
236 }

243 void VideoManager::ToggleSaveOutput()
244 {
245 m_saveOutput = !m_saveOutput;
246
247 if (m_saveOutput)
248 {
249 const std::string OUTPUT_FILE_NAME = std::format("{:%Y-%m-%d-%H-%M-%S}-output.mp4", std::chrono::system_clock::now());
250
251 m_outputVideo.open(OUTPUT_FILE_NAME,
252 cv::VideoWriter::fourcc('m', 'p', '4', 'v'),
253 Globals::G_OUTPUT_VIDEO_FPS,
254 cv::Size(Globals::G_OUTPUT_VIDEO_WIDTH, Globals::G_OUTPUT_VIDEO_HEIGHT));
255
256 if (!m_outputVideo.isOpened())
257 {
258 std::cout << std::format("\nERROR: Output video file '{}' could not be opened! Recording stopped!\n", OUTPUT_FILE_NAME);
259 m_saveOutput = false;
260 }
261
262 else
263 {
264 m_videoManagerInformation.m_saveOutputText = Globals::G_UI_TEXT_RECORDING;
265 }
266 }
267
268 else
269 {
270 m_outputVideo.release();
271 m_videoManagerInformation.m_saveOutputText = Globals::G_UI_TEXT_NOT_RECORDING;
272 }
