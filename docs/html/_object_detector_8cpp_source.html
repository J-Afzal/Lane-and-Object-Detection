22 void ObjectDetector::SetProperties(const std::string& p_yoloFolderPath,
23 const Globals::ObjectDetectorTypes& p_objectDetectorTypes,
24 const Globals::ObjectDetectorBackEnds& p_objectDetectorBackEnds,
25 const Globals::ObjectDetectorBlobSizes& p_objectDetectorBlobSizes)
26 {
27 switch (p_objectDetectorTypes)
28 {
29 case Globals::ObjectDetectorTypes::NONE:
30 m_skipObjectDetection = true;
31 break;
32
33 case Globals::ObjectDetectorTypes::STANDARD:
34 m_net = cv::dnn::readNetFromDarknet(p_yoloFolderPath + "yolov4.cfg", p_yoloFolderPath + "yolov4.weights");
35 break;
36
37 case Globals::ObjectDetectorTypes::TINY:
38 m_net = cv::dnn::readNetFromDarknet(p_yoloFolderPath + "yolov4-tiny.cfg", p_yoloFolderPath + "yolov4-tiny.weights");
39 break;
40
41 default:
42 throw Globals::Exceptions::NotImplementedError();
43 break;
44 }
45
46 switch (p_objectDetectorBackEnds)
47 {
48 case Globals::ObjectDetectorBackEnds::CPU:
49 m_net.setPreferableBackend(cv::dnn::DNN_BACKEND_OPENCV);
50 m_net.setPreferableTarget(cv::dnn::DNN_TARGET_CPU);
51 break;
52
53 case Globals::ObjectDetectorBackEnds::CUDA:
54 m_net.setPreferableBackend(cv::dnn::DNN_BACKEND_CUDA);
55 m_net.setPreferableTarget(cv::dnn::DNN_TARGET_CUDA);
56 break;
57
58 default:
59 throw Globals::Exceptions::NotImplementedError();
60 break;
61 }
62
63 m_blobSize = static_cast<int32_t>(p_objectDetectorBlobSizes);
64
65 m_unconnectedOutputLayerNames = m_net.getUnconnectedOutLayersNames();
66 }

68 void ObjectDetector::RunObjectDetector(const cv::Mat& p_frame)
69 {
70 if (m_skipObjectDetection)
71 {
72 return;
73 }
74
75 m_objectDetectionInformation.m_objectInformation.clear();
76
77 // Get output blobs from the frame
78 std::vector<cv::Mat> outputBlobs;
79 const cv::Mat BLOB_FROM_IMAGE = cv::dnn::blobFromImage(p_frame, Globals::G_OBJECT_DETECTOR_SCALE_FACTOR, cv::Size(m_blobSize, m_blobSize), Globals::G_COLOUR_BLACK);
80 m_net.setInput(BLOB_FROM_IMAGE);
81 m_net.forward(outputBlobs, m_unconnectedOutputLayerNames);
82
83 // Go through all output blobs and only allow those with confidence above threshold
84 std::vector<std::string> initialObjectNames;
85 std::vector<cv::Rect> initialObjectBoundingBoxes;
86 std::vector<float> initialObjectConfidences;
87 cv::Point maxConfidenceObjectIndex;
88 double maxConfidence = -DBL_MAX;
89
90 for (auto& outputBlob : outputBlobs)
91 {
92 for (int32_t rowIndex = 0; rowIndex < outputBlob.rows; rowIndex++)
93 {
94 // Each row is structured as the following [cx, cy, w, h, box score, object 1 score, object 2 score, ..., object n score]
95
96 // minMaxLoc returns the object index that has the maximum confidence as well as the maximum confidence
97 cv::minMaxLoc(outputBlob.row(rowIndex).colRange(Globals::G_OBJECT_DETECTOR_OUTPUT_BLOBS_OBJECT_SCORES_START_INDEX, outputBlob.cols), nullptr, &maxConfidence, nullptr, &maxConfidenceObjectIndex);
98
99 if (maxConfidence > Globals::G_OBJECT_DETECTOR_CONFIDENCE_THRESHOLD)
100 {
101 const double CENTER_X = outputBlob.at<float>(rowIndex, Globals::G_OBJECT_DETECTOR_OUTPUT_BLOBS_CENTER_X_COORD_INDEX) * Globals::G_INPUT_VIDEO_WIDTH;
102 const double CENTER_Y = outputBlob.at<float>(rowIndex, Globals::G_OBJECT_DETECTOR_OUTPUT_BLOBS_CENTER_Y_COORD_INDEX) * Globals::G_INPUT_VIDEO_HEIGHT;
103 const double WIDTH = (outputBlob.at<float>(rowIndex, Globals::G_OBJECT_DETECTOR_OUTPUT_BLOBS_WIDTH_INDEX) * Globals::G_INPUT_VIDEO_WIDTH) + Globals::G_OBJECT_DETECTOR_BOUNDING_BOX_BUFFER;
104 const double HEIGHT = (outputBlob.at<float>(rowIndex, Globals::G_OBJECT_DETECTOR_OUTPUT_BLOBS_HEIGHT_INDEX) * Globals::G_INPUT_VIDEO_HEIGHT) + Globals::G_OBJECT_DETECTOR_BOUNDING_BOX_BUFFER;
105
106 // Remove object detections on the hood of car
107 if (CENTER_Y < Globals::G_ROI_BOTTOM_HEIGHT)
108 {
109 const double DIVISOR = 2; // TODO(Main): const
110 initialObjectNames.push_back(Globals::G_OBJECT_DETECTOR_OBJECT_NAMES.at(maxConfidenceObjectIndex.x));
111 initialObjectBoundingBoxes.emplace_back(CENTER_X - (WIDTH / DIVISOR), CENTER_Y - (HEIGHT / DIVISOR), WIDTH, HEIGHT);
112 initialObjectConfidences.push_back(static_cast<float>(maxConfidence));
113 }
114 }
115 }
116 }
117
118 // Apply non-maxima suppression to suppress overlapping bounding boxes for objects that overlap, the highest confidence object will be chosen
119 std::vector<int> nonMaximaSuppressedFilteredIndicies;
120
121 cv::dnn::NMSBoxes(initialObjectBoundingBoxes, initialObjectConfidences, 0.0, static_cast<float>(Globals::G_OBJECT_DETECTOR_NMS_THRESHOLD), nonMaximaSuppressedFilteredIndicies);
122
123 const uint32_t CONVERT_TO_PERCENTAGE = 100; // TODO(Main): const
124
125 for (const int32_t& index : nonMaximaSuppressedFilteredIndicies)
126 {
127 m_objectDetectionInformation.m_objectInformation.push_back({.m_boundingBox = initialObjectBoundingBoxes.at(index),
128 .m_boundingBoxColour = Globals::G_OBJECT_DETECTOR_OBJECT_NAMES_AND_COLOURS.at(initialObjectNames.at(index)),
129 .m_objectName = std::format("{} ({} %)", initialObjectNames.at(index), std::to_string(static_cast<uint32_t>(initialObjectConfidences.at(index) * CONVERT_TO_PERCENTAGE)))});
130 }
131 }
