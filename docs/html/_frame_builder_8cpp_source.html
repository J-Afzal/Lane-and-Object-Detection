33 void FrameBuilder::AddObjectDetectorInformation(cv::Mat& p_frame, const ObjectDetectionInformation& p_objectDetectionInformation)
34 {
35 for (const ObjectDetectionInformation::DetectedObjectInformation& objectInformation : p_objectDetectionInformation.m_objectInformation)
36 {
37 // Draw the bounding box
38 cv::rectangle(p_frame,
39 objectInformation.m_boundingBox,
40 objectInformation.m_boundingBoxColour,
41 Globals::G_OBJECT_DETECTOR_BOUNDING_BOX_BORDER_THICKNESS,

43

50 objectInformation.m_boundingBoxColour,
51 cv::FILLED,
52 cv::LINE_AA);
53
54 // Draw the bounding box header text

56 objectInformation.m_objectName,
57 cv::Point(objectInformation.m_boundingBox.x, objectInformation.m_boundingBox.y - Globals::G_OBJECT_DETECTOR_BOUNDING_BOX_TEXT_HEIGHT_OFFSET),
58 Globals::G_DEFAULT_FONT_FACE,

62 cv::LINE_AA);

66 void FrameBuilder::AddLaneDetectorInformation(cv::Mat& p_frame, const LaneDetectionInformation& p_laneDetectionInformation)
67 {
68 // Add driving state
69 AddBackgroundRectAndCentredText(p_frame, Globals::G_UI_DRIVING_STATE_RECT, p_laneDetectionInformation.m_drivingStateTitle);
70
71 // Add turning state
72 AddBackgroundRectAndCentredText(p_frame, Globals::G_UI_TURNING_STATE_RECT, p_laneDetectionInformation.m_turningStateTitle);
73
74 // Add lane information title
75 AddBackgroundRectAndCentredText(p_frame, Globals::G_UI_LANE_INFORMATION_TITLE_RECT, p_laneDetectionInformation.m_laneInformationTitle);
76
77 // Add lane information lane lines states
78 cv::rectangle(p_frame, Globals::G_UI_LANE_INFORMATION_RECT, Globals::G_COLOUR_BLACK, cv::FILLED, cv::LINE_AA);
79
80 // Left line states
81 for (int32_t i = 0; i < p_laneDetectionInformation.m_leftLineTypesForDisplay.size(); i++)
82 {
83 cv::rectangle(p_frame,
84 cv::Rect(Globals::G_LANE_INFORMATION_LEFT_LANE_STATE_X_LOCATION,
85 Globals::G_LANE_INFORMATION_LANE_STATE_Y_START_LOCATION + (i * Globals::G_LANE_INFORMATION_LANE_STATE_HEIGHT * 2),
86 Globals::G_LANE_INFORMATION_LANE_STATE_WIDTH,
87 static_cast<int>(Globals::G_LANE_INFORMATION_LANE_STATE_HEIGHT * p_laneDetectionInformation.m_leftLineTypesForDisplay[i])),
88 Globals::G_COLOUR_WHITE,
89 cv::FILLED,
90 cv::LINE_AA);
91 }
92
93 // Middle line states
94 for (uint32_t i = 0; i < p_laneDetectionInformation.m_middleLineTypesForDisplay.size(); i++)
95 {
96 cv::rectangle(p_frame,
97 cv::Rect(Globals::G_LANE_INFORMATION_MIDDLE_LANE_STATE_X_LOCATION,
98 Globals::G_LANE_INFORMATION_LANE_STATE_Y_START_LOCATION + static_cast<int>(i * Globals::G_LANE_INFORMATION_LANE_STATE_HEIGHT * 2),
99 Globals::G_LANE_INFORMATION_LANE_STATE_WIDTH,
100 static_cast<int>(Globals::G_LANE_INFORMATION_LANE_STATE_HEIGHT * p_laneDetectionInformation.m_middleLineTypesForDisplay[i])),
101 Globals::G_COLOUR_WHITE,
102 cv::FILLED,
103 cv::LINE_AA);
104 }
105
106 // Right line states
107 for (int32_t i = 0; i < p_laneDetectionInformation.m_rightLineTypesForDisplay.size(); i++)
108 {
109 cv::rectangle(p_frame,
110 cv::Rect(Globals::G_LANE_INFORMATION_RIGHT_LANE_STATE_X_LOCATION,
111 Globals::G_LANE_INFORMATION_LANE_STATE_Y_START_LOCATION + (i * Globals::G_LANE_INFORMATION_LANE_STATE_HEIGHT * 2),
112 Globals::G_LANE_INFORMATION_LANE_STATE_WIDTH,
113 static_cast<int>(Globals::G_LANE_INFORMATION_LANE_STATE_HEIGHT * p_laneDetectionInformation.m_rightLineTypesForDisplay[i])),
114 Globals::G_COLOUR_WHITE,
115 cv::FILLED,
116 cv::LINE_AA);
117 }
118
119 // Add turning required to return to center of the lane
120 AddBackgroundRectAndCentredText(p_frame, Globals::G_UI_LANE_INFORMATION_TURNING_REQUIRED_RECT, p_laneDetectionInformation.m_turningRequiredToReturnToCenterText);
121
122 // Used for the below overlays
123 cv::Mat blankFrame = cv::Mat::zeros(Globals::G_INPUT_VIDEO_HEIGHT, Globals::G_INPUT_VIDEO_WIDTH, p_frame.type());
124
125 if (p_laneDetectionInformation.m_drivingState == Globals::DrivingState::WITHIN_LANE)
126 {
127 // Add the box overlay signifying vehicle location within the lane
128 cv::rectangle(blankFrame,
129 cv::Rect(Globals::G_LANE_INFORMATION_VEHICLE_POSITION_X_MIDDLE_LOCATION - p_laneDetectionInformation.m_turningRequiredToReturnToCenterPercentage,
130 Globals::G_LANE_INFORMATION_VEHICLE_POSITION_Y_LOCATION,
131 Globals::G_LANE_INFORMATION_VEHICLE_POSITION_WIDTH,
132 Globals::G_LANE_INFORMATION_VEHICLE_POSITION_HEIGHT),
133 Globals::G_COLOUR_YELLOW,
134 cv::FILLED,
135 cv::LINE_AA);
136 }
137
138 // Draw the green lane overlay to signify the area of the road which is considered the 'current lane'
139 cv::fillConvexPoly(blankFrame, p_laneDetectionInformation.m_laneOverlayCorners, Globals::G_LANE_INFORMATION_LANE_OVERLAY_COLOUR, cv::LINE_AA);
140
141 // Using add to give a transparent overlay
142 cv::add(p_frame, blankFrame, p_frame);
143 }

172 void FrameBuilder::AddCentredText(cv::Mat& p_frame, const cv::Rect& p_backgroundRect, const std::string& p_text, const double& p_fontScale)
173 {
174 double currentFontScale = p_fontScale;
175
176 cv::Size textSize = cv::getTextSize(p_text, Globals::G_DEFAULT_FONT_FACE, currentFontScale, Globals::G_DEFAULT_FONT_THICKNESS, nullptr);
177
178 while ((((textSize.width + Globals::G_DEFAULT_HORIZONTAL_PADDING) > p_backgroundRect.width) ||
179 ((textSize.height + Globals::G_DEFAULT_VERTICAL_PADDING) > p_backgroundRect.height)) &&
180 (currentFontScale > Globals::G_DEFAULT_FONT_DECREMENT))
181 {
182 currentFontScale -= Globals::G_DEFAULT_FONT_DECREMENT;
183 textSize = cv::getTextSize(p_text, Globals::G_DEFAULT_FONT_FACE, currentFontScale, Globals::G_DEFAULT_FONT_THICKNESS, nullptr);
184 }
185
186 const int32_t X_PAD = std::ceil((p_backgroundRect.width - textSize.width) / Globals::G_DIVIDE_BY_TWO);
187 const int32_t Y_PAD = std::ceil((p_backgroundRect.height - textSize.height) / Globals::G_DIVIDE_BY_TWO);
188
189 cv::putText(p_frame,
190 p_text,
191 {p_backgroundRect.x + X_PAD, p_backgroundRect.y + textSize.height + Y_PAD},
192 Globals::G_DEFAULT_FONT_FACE,
193 currentFontScale,
194 Globals::G_COLOUR_WHITE,
195 Globals::G_DEFAULT_FONT_THICKNESS,
196 cv::LINE_AA);
197 }
